mqtt:
  sensor:
    - name: "HAN bridge RSSI"
      state_topic: "pulse/publish"
      # Vi setter encoding til tom for å hindre at HA krasjer på binære data
      encoding: "" 
      value_template: >
        {% set val = value.decode('utf-8', 'ignore') %}
        {% if 'status' in val and 'rssi' in val %}
          {{ (val | from_json).status.rssi }}
        {% else %}
          {{ states('sensor.han_bridge_rssi') }}
        {% endif %}
      unit_of_measurement: "dBm"
      device_class: "signal_strength"

    - name: "HAN bridge temperatur (NTC)"
      state_topic: "pulse/publish"
      encoding: ""
      value_template: >
        {% set val = value.decode('utf-8', 'ignore') %}
        {% if 'status' in val and 'ntc' in val %}
          {{ (val | from_json).status.ntc }}
        {% else %}
          {{ states('sensor.han_bridge_temperatur_ntc') }}
        {% endif %}
      unit_of_measurement: "°C"
      device_class: "temperature"
